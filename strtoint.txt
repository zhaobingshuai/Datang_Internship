#include<stdio.h>
#include<limits.h>
#include<iostream>
using namespace std;
int strtoint(const char*str)
{
		if(str == NULL || *str == '\0')
		{
				return 0;
		}
		long long num = 0;
		int sign = 1;
		while(*str == ' ')
		{
		    str++;
		}
		if(*str == '+')
		{
		    str++;
		}
		else if(*str == '-')
		{
		    str++;
			sign = -1;
		}
		while(*str != '\0')
		{
				if(*str >= '0'  && *str <= '9')
				{
						num = num *10 + (*str - '0')*sign;
						if(num >= INT_MAX)
						{
								return INT_MAX;
						}
						else if(num <= INT_MIN)
						{
								return INT_MIN;
						}
				}
				else
				{
						break;
				}
				str++;
		}
		return (int)num;
}
